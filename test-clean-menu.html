تنظيف القائمة من الترقيم القديم

حذف جميع الفراغات والأرقام القديمة من القائمة المحمولة

عملية التنظيف

  • الهدف: حذف جميع الأرقام والفراغات القديمة من القائمة
  • التنظيف: إزالة عناصر .menu-number والأرقام في النصوص
  • الفراغات: تنظيف الفراغات الزائدة من الأزرار
  • النتيجة: قائمة نظيفة بدون أي ترقيم

كود التنظيف

🧹 دالة التنظيف الرئيسية:

function cleanOldNumberingFromMenu() { console.log('🧹 تنظيف الترقيم والفراغات القديمة من القائمة المحمولة...'); const menuList = document.querySelector('.mobile-dropdown-list'); const allButtons = menuList.querySelectorAll('li > button'); let cleanedCount = 0; allButtons.forEach(button => { // إزالة عناصر الأرقام المضافة بـ JavaScript const existingNumbers = button.querySelectorAll('.menu-number'); existingNumbers.forEach(num => { num.remove(); cleanedCount++; }); // إزالة أي أرقام في بداية النص const textNodes = Array.from(button.childNodes) .filter(node => node.nodeType === Node.TEXT_NODE); textNodes.forEach(textNode => { const cleanedText = textNode.textContent .replace(/^\s*\d+\.\s*/, '').trim(); if (cleanedText !== textNode.textContent.trim()) { textNode.textContent = cleanedText; cleanedCount++; } }); // تنظيف الفراغات الزائدة button.innerHTML = button.innerHTML .replace(/^\s+|\s+$/g, '') .replace(/\s+/g, ' '); }); console.log(`✅ تم تنظيف ${cleanedCount} عنصر`); }

⚡ دالة التنظيف المبسطة:

function cleanMenuOnly() { console.log('🧹 تنظيف القائمة المحمولة من الترقيم والفراغات القديمة...'); cleanOldNumberingFromMenu(); console.log('✅ تم تنظيف القائمة بنجاح'); } // استدعاء عند فتح القائمة mobileMenuBtn.addEventListener('click', function() { // ... كود فتح القائمة // تنظيف القائمة من الترقيم والفراغات القديمة فقط cleanMenuOnly(); });

اختبار التنظيف

خطوات الاختبار

📋 للاختبار الفعلي:

  1. افتح الصفحة الرئيسية (index.html)
  2. افتح Console (F12) لمراقبة عملية التنظيف
  3. قلل حجم الشاشة لأقل من 900px
  4. انقر على زر القائمة (☰) في الهيدر
  5. تحقق من النتائج:
    • ✅ في Console: "🧹 تنظيف القائمة المحمولة..."
    • ✅ في Console: "✅ تم تنظيف X عنصر"
    • ✅ الأزرار نظيفة بدون أرقام
    • ✅ لا توجد فراغات زائدة
    • ✅ نصوص الأزرار واضحة ونظيفة
  6. اختبر عدة مرات:
    • أغلق القائمة وافتحها مرة أخرى
    • تأكد من عدم تراكم الأرقام
    • تأكد من نظافة القائمة في كل مرة

النتائج المتوقعة

🎯 ما ستراه:

  • قائمة نظيفة: بدون أي أرقام أو ترقيم
  • نصوص واضحة: أسماء الأزرار فقط
  • بدون فراغات: لا توجد فراغات زائدة
  • أيقونات سليمة: الأيقونات تظهر بشكل طبيعي
  • تنظيف تلقائي: يحدث عند كل فتح للقائمة
